━━━━━━━━━━━━━━━━━━━━━━
🧠 1) Question Engine (محرك الأسئلة)
━━━━━━━━━━━━━━━━━━━━━━

ده قلب اللعبة.

بيتحكم في:
- نوع السؤال
- صعوبته
- توقيته
- مصدره (فيديو / صورة / صوت / نص)

━━━━━━━━━━━━━━━━━━━━━━
⚙️ بيتعمل إزاي؟

Database:
- Questions Table
- Media Links
- Difficulty Level
- Tags (Science, History…)

Backend:
- اختيار السؤال حسب اللاعب
- Randomization
- Anti-repeat system

Frontend:
- عرض السؤال بشكل سينمائي
- Timer + Animation

━━━━━━━━━━━━━━━━━━━━━━
📊 2) Difficulty Scaling System
━━━━━━━━━━━━━━━━━━━━━━

اللعبة لازم تتأقلم مع اللاعب.

لو اللاعب قوي:
→ الأسئلة أصعب

لو ضعيف:
→ أسهل + شرح أكتر

━━━━━━━━━━━━━━━━━━━━━━
🧠 بيتعمل إزاي؟
━━━━━━━━━━━━━━━━━━━━━━

- Track Performance
- Win/Loss Ratio
- Response Time
- Accuracy

ثم AI/Algorithm يحدد المستوى.

━━━━━━━━━━━━━━━━━━━━━━
🎯 3) Challenge System

مش مجرد أسئلة.

أنواع التحديات:

- Video Analysis

━━━━━━━━━━━━━━━━━━━━━━

مثال:
Video Challenge Module:
- video URL
- questions after playback
